# React-formify
this package help you to create custom form with js object


"# react-formify" 





#example object
const formify = [
  {
    node: 'input',
    type: 'text',
    label: 'name',
    props: {
      style: {
        fontSize: '40px'
      }
    }
  }, 
  {
    node: 'select',
    label: 'gradee',
    options: ['one', 'two', 'three']
  }
];
